Problem Link: https://leetcode.com/problems/valid-parentheses

Solution in Java:

```
class Solution {
    public boolean isValid(String s) {

        List<Integer> brackets = new ArrayList<>();

        for (int i = 0; i < s.length(); i++) {
            if (s.charAt(i) == '(') {
                brackets.add(1);
            } else if (s.charAt(i) == '[') {
                brackets.add(2);
            } else if (s.charAt(i) == '{') {
                brackets.add(3);
            } else if (s.charAt(i) == ')') {
                if (brackets.size() == 0) {
                    return false;
                }
                int lastItem = brackets.remove(brackets.size() - 1);
                if (lastItem != 1) {
                    return false;
                }
            } else if (s.charAt(i) == ']') {
                if (brackets.size() == 0) {
                    return false;
                }
                int lastItem = brackets.remove(brackets.size() - 1);
                if (lastItem != 2) {
                    return false;
                }
            } else {
                if (brackets.size() == 0) {
                    return false;
                }
                int lastItem = brackets.remove(brackets.size() - 1);
                if (lastItem != 3) {
                    return false;
                }
            }
        }

        if (brackets.size() > 0) {
            return false;
        }

        return true;

    }
}
```
