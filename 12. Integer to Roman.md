Problem Link: https://leetcode.com/problems/integer-to-roman/

Solution in Java:

```
class Solution {
    public String intToRoman(int num) {

        String result = "";

        if (num >= 4000) {
            return "Please pass a number less than 4000";
        } 
        
        if (num <= 0) {
            return "Number can not be negative";
        }

        while (num >= 1000) {
            result += "M";
            num -= 1000;
        }

        if (num < 1000 && num >= 900) {
            result += "CM";
            num -= 900;
        }

        while (num >= 500) {
            result += "D";
            num -= 500;
        }

        if (num < 500 && num >= 400) {
            result += "CD";
            num -= 400;
        }

        while (num >= 100) {
            result += "C";
            num -= 100;
        }

        if (num < 100 && num >=90) {
            result += "XC";
            num -= 90;
        }

        while (num >= 50) {
            result += "L";
            num -= 50;
        }

        if (num < 50 && num >= 40) {
            result += "XL";
            num -= 40;
        }

        while (num >= 10) {
            result += "X";
            num -= 10;
        }

        if (num == 9) {
            result += "IX";
            num -= 9;
        }

        while (num >= 5) {
            result += "V";
            num -= 5;
        }

        if (num == 4) {
            result += "IV";
            num -= 4;
        }

        while (num > 0) {
            result += "I";
            num -= 1;
        }

        return result;
        
    }
}
```
