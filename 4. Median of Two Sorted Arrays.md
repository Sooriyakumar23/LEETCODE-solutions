Solution in Java:

```
class Solution {
    public double findMedianSortedArrays(int[] nums1, int[] nums2) {

        List<Integer> nums1List = new ArrayList<>(Arrays.stream(nums1)
                .boxed()
                .toList());
        List<Integer> nums2List = new ArrayList<>(Arrays.stream(nums2)
                .boxed()
                .toList());

        for (int x : nums2List) {
            int index = indexFinder(nums1List, x);
            nums1List.add(index, x);
        }

        System.out.println(nums1List);

        if (nums1List.size() % 2 == 1) {
            return nums1List.get(nums1List.size() / 2);
        }

        int prevIndex = nums1List.size() / 2;
        int nextIndex = prevIndex - 1;

        return (nums1List.get(prevIndex) + nums1List.get(nextIndex)) / 2.0;

    }

    public int indexFinder(List<Integer> myList, int myNum) {
        int left = 0;
        int right = myList.size() - 1;

        while (left <= right) {
            int mid = left + (right - left) / 2;

            if (myList.get(mid) == myNum) {
                return mid;
            } else if (myList.get(mid) < myNum) {
                left = mid + 1;
            } else {
                right = mid - 1;
            }
        }

        return left;
    }
}
```
