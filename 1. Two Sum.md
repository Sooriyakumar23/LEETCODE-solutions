Problem Link: https://leetcode.com/problems/two-sum

Solution in Java:

```
class Solution {

public int[] twoSum(int[] nums, int target) {

        int[] answer = new int[2];

        Map<Integer, Integer> remainingValueWiseIndex = new HashMap<>();

        for (int i=0; i<nums.length; i++) {
            // System.out.println(remainingValueWiseIndex);
            if (remainingValueWiseIndex.containsKey(nums[i])) {
                answer[0] = remainingValueWiseIndex.get(nums[i]);
                answer[1] = i;

                return answer;
            }
            remainingValueWiseIndex.put(target-nums[i], i);
        }

        System.out.println(remainingValueWiseIndex);

        return answer;

    }

}
```
