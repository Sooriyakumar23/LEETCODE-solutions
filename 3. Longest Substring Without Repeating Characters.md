Problem Link: https://leetcode.com/problems/longest-substring-without-repeating-characters

Solution in Java:

```
class Solution {
    public int lengthOfLongestSubstring(String s) {

        int prevMaxLength = 0;

        Map<Character, Integer> characterIndexMap = new HashMap<>();
        for (int i = 0; i < s.length(); i++) {
            if (characterIndexMap.get(s.charAt(i)) != null) {
                if (prevMaxLength < characterIndexMap.size()) {
                    prevMaxLength = characterIndexMap.size();
                }
                int duplicateCharIndex = characterIndexMap.get(s.charAt(i));

                Iterator<Map.Entry<Character, Integer>> iterator = characterIndexMap.entrySet().iterator();
                while (iterator.hasNext()) {
                    Map.Entry<Character, Integer> entry = iterator.next();
                    if (entry.getValue() <= duplicateCharIndex) {
                        iterator.remove();
                    }
                }
            }
            characterIndexMap.put(s.charAt(i), i);
        }

        if (prevMaxLength < characterIndexMap.size()) {
            return characterIndexMap.size();
        } else {
            return prevMaxLength;
        }

    }
}
```
